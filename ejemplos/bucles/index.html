Bucles/Ciclos

Variables Condicionales Bucles Array Muestra

Bucle For

for (let i = 0; i < 5; i++) {
    console.log("Vuelta número: " + i);
}
// Vuelta número: 0
// Vuelta número: 1
// Vuelta número: 2
// Vuelta número: 3
// Vuelta número: 4

Bucle While

let contador = 0;
while (contador < 10) {
    console.log("Vuelta usando while: " + contador);
    contador++;
}
// Vuelta usando while: 0
// Vuelta usando while: 1
// Vuelta usando while: 2
// Vuelta usando while: 3
// Vuelta usando while: 4
// Vuelta usando while: 5
// Vuelta usando while: 6
// Vuelta usando while: 7
// Vuelta usando while: 8
// Vuelta usando while: 9

Bucle Do while

contador = 0;
do {
    console.log("Vuelta usando do while: " + contador);
    contador++;
} while (contador < 10)
